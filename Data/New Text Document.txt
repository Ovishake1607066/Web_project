long vcc=12,gnd=11,in=10;
void setup() {
  // put your setup code here, to run once:
  pinMode(vcc,OUTPUT);
  pinMode(gnd,OUTPUT);
  pinMode(in,OUTPUT);
  digitalWrite(vcc,HIGH);
  digitalWrite(gnd,LOW);
  

}

void loop() {
  // put your main code here, to run repeatedly:
  digitalWrite(in,HIGH);
  delay(1000);
  digitalWrite(in,LOW);
  delay(1000);

}
